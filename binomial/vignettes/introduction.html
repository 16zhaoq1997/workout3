Bin Distribution

#examples include:
#bin_distribution(trials = 5, prob = 0.5)
bin_distribution(trials = 5, prob = 0.5)
#>   success probability
#> 1       0     0.03125
#> 2       1     0.15625
#> 3       2     0.31250
#> 4       3     0.31250
#> 5       4     0.15625
#> 6       5     0.03125

Plot Bin Distributon

#You can also plot the above probability distribution
dis1 <- bin_distribution(trials = 5, prob = 0.5)
plot(dis1)

Bin Cumulative

This is a function that gives you the cumulative distribution table of the binomial random variable, given a probability and number of trials.

#examples include:
#bin_cumulative(trials = 5, prob = 0.5)
bin_cumulative(trials = 5, prob = 0.5)
#>   success probability cumulative
#> 1       0     0.03125    0.03125
#> 2       1     0.15625    0.18750
#> 3       2     0.31250    0.50000
#> 4       3     0.31250    0.81250
#> 5       4     0.15625    0.96875
#> 6       5     0.03125    1.00000

Plot Bin Cumulative

dis2 <- bin_cumulative(trials = 5, prob = 0.5)
plot(dis2)

Bin Probability.

#examples include:
#bin_probability(success = 2, trials = 5, prob = 0.5)
#bin_probability(success = 0:2, trials = 5, prob = 0.5)
#bin_probability(success = 55, trials = 100, prob = 0.45)

#> [1] 0.3125
bin_probability(success = 0:2, trials = 5, prob = 0.5)
#> Warning in if (success < 0 | success != round(success)) {: the condition
#> has length > 1 and only the first element will be used
#> Warning in if (success > trials) {: the condition has length > 1 and only
#> the first element will be used
#> Warning in if (k > n) {: the condition has length > 1 and only the first
#> element will be used
#> [1] 0.03125 0.15625 0.31250
bin_probability(success = 55, trials = 100, prob = 0.45)
#> [1] 0.01075277
